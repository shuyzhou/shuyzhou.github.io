Vue源码阅读（二）：数据监听

2017-06-17

###

Vue.prototype._proxy = function (key) {
if (!isReserved(key)) {
var self = this
Object.defineProperty(self, key, {
configurable: true,
enumerable: true,
get: function proxyGetter () {
return self._data[key]
},
set: function proxySetter (val) {
self._data[key] = val
}
})
}
}

function Observer (value) {
this.value = value
this.dep = new Dep()
def(value, ‘ob‘, this)
if (isArray(value)) {
var augment = hasProto
? protoAugment
: copyAugment
augment(value, arrayMethods, arrayKeys)
this.observeArray(value)
} else {
this.walk(value)
}
}

function defineReactive (obj, key, val) {
var dep = new Dep()

var property = Object.getOwnPropertyDescriptor(obj, key)
if (property && property.configurable === false) {
return
}

// cater for pre-defined getter/setters
var getter = property && property.get
var setter = property && property.set
//递归监听该属性的值
var childOb = observe(val)
Object.defineProperty(obj, key, {
enumerable: true,
configurable: true,
get: function reactiveGetter () {
var value = getter ? getter.call(obj) : val
//这里省略了对dep的操作
return value
},
set: function reactiveSetter (newVal) {
var value = getter ? getter.call(obj) : val
if (newVal === value) {
return
}
if (setter) {
setter.call(obj, newVal)
} else {
val = newVal
}
childOb = observe(newVal)
dep.notify()
}
})
}
`` defineReactive方法重新定义每个属性的getter和setter，并对该属性的值进行监听。在observe方法中，对这个值又new了一个新的Observer实例并返回，这样就实现了对数据的深层次的监听。在setter中，同样对新的值进行监听，并且利用notify方法发送消息给与该属性相关的directive`。

$data属性

$data属性定义在Vue的原型上的，Vue利用defineProperty方法重新定义了$data的getter和setter，当$data被访问时返回当前vue实例的_data属性，该实例的数据实际上是保存在这个属性里，而给$data赋值时则会调用实例方法_setData。 在_setData方法中，将新的data赋值给实例的_data属性。比较新的data和旧的data，对需要删除的属性解除代理，对于需要新增的属性绑定代理。接着，断开旧data对该实例的引用，调用observe方法对新的data进行监听，最后调用实例的_digest方法强制更新所有watcher`。
